const services = [
    { service: "Eye Check-up", price: "₹500 – ₹700" },
    { service: "LASIK Surgery", price: "₹45,000 – ₹60,000" },
    { service: "Cataract Surgery", price: "₹25,000 – ₹40,000" }
];

const hospitalGrid = document.getElementById("hospitalGrid");
const priceCards = document.getElementById("priceCards");
const modal = document.getElementById("appointmentModal");

function loadHospitals(list = hospitals) {
    hospitalGrid.innerHTML = "";
    list.forEach(h => {
        hospitalGrid.innerHTML += `
            <div class="hospital-card">
                <h3>${h.name}</h3>
                <p>${h.location}</p>
                <p>⭐ ${h.rating}</p>
                <button onclick="openModal()">Book Appointment</button>
            </div>
        `;
    });
}

function loadPrices() {
    services.forEach(s => {
        priceCards.innerHTML += `
            <div class="price-card">
                <h4>${s.service}</h4>
                <p>${s.price}</p>
            </div>
        `;
    });
}

document.getElementById("searchInput").addEventListener("input", e => {
    const value = e.target.value.toLowerCase();
    loadHospitals(hospitals.filter(h => h.name.toLowerCase().includes(value)));
});

function openModal() {
    modal.style.display = "block";
}

function closeModal() {
    modal.style.display = "none";
}

function submitAppointment() {
    document.getElementById("confirmMsg").textContent =
        "Appointment request submitted successfully ✔";
}

function scrollToHospitals() {
    document.getElementById("hospitals").scrollIntoView({ behavior: "smooth" });
}

loadHospitals();
loadPrices();
